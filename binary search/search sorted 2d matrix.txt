class Solution {
    public boolean searchMatrix(int[][] matrix, int target) {
        int left=0;
        int row=matrix.length;
        int col=matrix[0].length;

        int right=row*col-1;

        while(left<=right){

            int mid = (left+right)/2;

            int r= mid/col;
            int c=mid%col;

            if(matrix[r][c]==target){
                return true;
            }
            else if(matrix[r][c]< target){
                left=mid+1;
            }
            else{
                right=mid-1;
            }
        }

        return false;
    }
}
